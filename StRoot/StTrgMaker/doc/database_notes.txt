Show tables in database:
  show tables ;

Show columns of table trgDescriptor:
  select name,type,comment from schema where structName='trgDescriptor' ;

Find min-bias runs:
  if runNumber < 1226000 
    database RunLog
    select runNumber from trgDescriptor where zdcLogic=3 
    and (ctbZdcLogic=0 or ctbZdcLogic=1);
  else if runNumber >= 1226000
    database RunLog_onl
    select runNumber,description from trgLogic where description like '%zdcE and zdcW%';
    or
    select runNumber,tcuFile from trgFile where tcuFile like '%zdcAND%';

Find min-bias runs (informal):
  database RunLog_onl
  select runNumber from daqSummary where comment like '%min%bias%' ;
